declare(strict_types=1);

namespace BaksDev\Ozon\Orders\Api\Exemplar;

use BaksDev\Ozon\Api\Ozon;
use DateInterval;
use Symfony\Component\DependencyInjection\Attribute\Autoconfigure;
use Symfony\Contracts\Cache\ItemInterface;

#[Autoconfigure(public: true)] /* TODO: удалить !!! */
final class GetOzonOrdersExemplarStatusRequest extends Ozon
{
    private array|false $exemplars = false;

    /**
     * Получить статус добавления экземпляров
     *
     * @see https://docs.ozon.ru/api/seller/#operation/PostingAPI_FbsPostingProductExemplarStatusV5
     *
     */
    public function find(string $number): bool|self
    {
        $number = str_replace('O-', '', $number);

        $cache = $this->getCacheInit('ozon-orders');
        $key = md5($number.self::class);

        return $cache->get($key, function(ItemInterface $item) use ($number): bool {

            /** Временно кешируем этикетку на 1 секунду */
            $item->expiresAfter(DateInterval::createFromDateString('3 seconds'));

            $response = $this->TokenHttpClient()
                ->request(
                    'POST',
                    '/v5/fbs/posting/product/exemplar/status',
                    ['json' => ['posting_number' => $number]],
                );

            $content = $response->toArray(false);

            if($response->getStatusCode() !== 200)
            {
                $this->logger->critical(
                    message: sprintf('ozon-orders: Ошибка при получении информации о статусе экземпляра отправления %s', $number),
                    context: [self::class.':'.__LINE__, $content],
                );

                return false;
            }

            if(true === isset($content['status']) && true === in_array($content['status'], ['ship_available', 'update_available']))
            {
                $item->expiresAfter(DateInterval::createFromDateString('1 day'));
                return true;
            }

            $this->exemplars = $content;

            return false;
        });
    }

    public function getExemplars(): array|false
    {
        return $this->exemplars;
    }
}

#[Group('ozon-orders')]

class GetOzonOrdersExemplarStatusRequestTest extends KernelTestCase
{
    private static OzonAuthorizationToken $Authorization;

    public static function setUpBeforeClass(): void
    {
        OrderNewTest::setUpBeforeClass();
        PackageProductStockTest::setUpBeforeClass();
        UserNewUserProfileHandleTest::setUpBeforeClass();

        self::$Authorization = new OzonAuthorizationToken(
            new UserProfileUid('018d464d-c67a-7285-8192-7235b0510924'),
            $_SERVER['TEST_OZON_TOKEN'],
            TypeProfileFbsOzon::TYPE,
            $_SERVER['TEST_OZON_CLIENT'],
            $_SERVER['TEST_OZON_WAREHOUSE'],
            '10',
            0,
            false,
            false,
        );
    }

    public function testUseCase(): void
    {
        self::assertTrue(true);

        /** @var GetOzonOrdersExemplarStatusRequest $GetOzonOrdersExemplarStatusRequest */
        $GetOzonOrdersExemplarStatusRequest = self::getContainer()->get(GetOzonOrdersExemplarStatusRequest::class);
        $GetOzonOrdersExemplarStatusRequest->TokenHttpClient(self::$Authorization);

        $orders = $GetOzonOrdersExemplarStatusRequest
            ->find('97272653-0194-1');

        if(true === $orders)
        {
            return;
        }

        /** Получаем информацию о честном знаке */

        foreach($orders['products'] as $prd => $order)
        {
            foreach($order['exemplars'] as $exm => $exemplar)
            {
                /** Передаем в качестве ГТД - комментарий   */
                $orders['products'][$prd]['exemplars'][$exm]['gtd'] = '10702070/040825/5266819';

                /** Передаем в качестве маркировки - честный знак   */
                $orders['products'][$prd]['exemplars'][$exm]['marks'] = [
                    'mark' => '0104650198060780215Y%hlED&Bigd,',
                    'mark_type' => 'mandatory_mark',
                ];
            }
        }

        /** @var UpdateOzonOrdersExemplarRequest $UpdateOzonOrdersExemplarRequest */
        $UpdateOzonOrdersExemplarRequest = self::getContainer()->get(UpdateOzonOrdersExemplarRequest::class);
        $UpdateOzonOrdersExemplarRequest->TokenHttpClient(self::$Authorization);

        $UpdateOzonOrdersExemplarRequest->update($orders);

        dd('$orders'); /* TODO: удалить !!! */


    }
}
